```jsx live="true" title="Navlink"
function Navlink({
  text = "Navlink",
  to = "/components",
  location = { pathname: "/" },
}) {
  return (
    <Link
      className="nav-link"
      data-is-active={location.pathname === to}
      to={to}
    >
      <h4>{text}</h4>
    </Link>
  )
}
```
